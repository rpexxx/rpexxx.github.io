function openExternalBrowser(url) { const isAndroid = /Android/i.test(navigator.userAgent); if (isAndroid) { // intent для Chrome (или браузера по умолчанию) const intentUrl = `intent://${url.replace(/^https?:\/\//, '')}#Intent;scheme=https;package=com.android.chrome;end`; window.location.href = intentUrl; // фолбек через 1 секунду (если интент не сработал) setTimeout(() => { window.location.href = url; }, 1000); } else { window.location.href = url; } } // использовать так: openExternalBrowser("https://onlyfans.com/ellablonde/c300");
